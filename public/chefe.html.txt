<script>
const SENHAS={Alfa:'alfa123',Bravo:'bravo123',Charlie:'charlie123',Delta:'delta123'};
const CHEFES={Alfa:'Cerqueira',Bravo:'Savio',Charlie:'Cleber',Delta:'Douglas'};
let equipeLogada=null,checklistAtual=null;

function fazerLogin(){
  const eq=document.getElementById('login-equipe').value;
  const pw=document.getElementById('login-senha').value;
  const err=document.getElementById('login-error');
  if(!eq){err.textContent='Selecione sua equipe.';err.style.display='block';return;}
  if(!pw){err.textContent='Digite sua senha.';err.style.display='block';return;}
  if(SENHAS[eq]!==pw){err.textContent='Senha incorreta.';err.style.display='block';return;}
  equipeLogada=eq;
  err.style.display='none';
  document.getElementById('tela-login').style.display='none';
  document.getElementById('tela-principal').style.display='block';
  document.getElementById('titulo-equipe').textContent='Equipe '+eq;
  document.getElementById('subtitulo-chefe').textContent='Chefe: '+CHEFES[eq]+' — Checklists para revisão';
  document.getElementById('hdr-info').textContent=eq+' — '+CHEFES[eq];
  document.getElementById('hdr-info').style.display='block';
  carregarChecklists();
}

async function carregarChecklists(){
  const lista=document.getElementById('lista-checklists');
  lista.innerHTML='<div style="text-align:center;padding:40px;color:var(--texto-sec);">Carregando...</div>';
  try{
    const r=await fetch('/api/checklists');
    const todos=await r.json();
    const meus=todos.filter(c=>c.equipe===equipeLogada);
    if(meus.length===0){lista.innerHTML='<div class="card" style="text-align:center;padding:40px;color:var(--texto-sec);">Nenhum checklist encontrado para sua equipe.</div>';return;}
    lista.innerHTML=meus.sort((a,b)=>new Date(b.data)-new Date(a.data)).map(c=>{
      const assinado=c.assinatura_chefe_aprovado;
      const badge=assinado?'<span class="badge badge-assinado">✅ Aprovado</span>':c.nc>0?'<span class="badge badge-nc">⚠ '+c.nc+' NC</span>':'<span class="badge badge-ok">✓ Conforme</span>';
      return '<div class="item" onclick="abrirChecklist(\''+c.id+'\')"><div style="display:flex;justify-content:space-between;align-items:center;flex-wrap:wrap;gap:8px;"><div><div style="font-size:14px;font-weight:700;">'+c.formulario_titulo+'</div><div style="font-size:12px;color:var(--texto-sec);margin-top:4px;">📅 '+c.data+' • 🚑 '+c.placa+' • 👤 '+(c.nome||'')+'</div></div>'+badge+'</div></div>';
    }).join('');
  }catch(e){lista.innerHTML='<div class="alert-e">Erro ao carregar checklists.</div>';}
}

async function abrirChecklist(id){
  const r=await fetch('/api/checklists');
  const todos=await r.json();
  checklistAtual=todos.find(c=>String(c.id)===String(id));
  if(!checklistAtual)return;
  document.getElementById('modal-titulo').textContent=checklistAtual.formulario_titulo;
  const itens=Object.values(checklistAtual.itens||{});
  const ncs=itens.filter(i=>i.status==='NC');
  const conf=itens.filter(i=>i.status==='C').length;
  let html='<div style="display:grid;grid-template-columns:1fr 1fr;gap:12px;margin-bottom:16px;font-size:13px;">'
    +'<div><strong>Viatura:</strong> '+checklistAtual.placa+' — '+checklistAtual.modelo+'</div>'
    +'<div><strong>Data:</strong> '+checklistAtual.data+'</div>'
    +'<div><strong>Preenchido por:</strong> '+(checklistAtual.nome||'')+'</div>'
    +'<div><strong>Equipe:</strong> '+(checklistAtual.equipe||'')+'</div>'
    +'</div>'
    +'<div style="display:flex;gap:10px;margin-bottom:16px;flex-wrap:wrap;">'
    +'<span class="badge badge-ok">✓ '+conf+' Conformes</span>'
    +'<span class="badge badge-nc">⚠ '+ncs.length+' NC</span>'
    +'</div>';
  if(ncs.length>0){
    html+='<div style="margin-bottom:16px;"><strong style="font-size:13px;">Itens Não Conformes:</strong><div style="margin-top:8px;">';
    ncs.forEach(nc=>{html+='<div class="nc-item">⚠ '+nc.num+' — '+nc.item+'</div>';});
    html+='</div></div>';
  }
  if(checklistAtual.obs){html+='<div style="margin-bottom:16px;"><strong style="font-size:13px;">Observações:</strong><div style="background:var(--cinza-claro);padding:10px;border-radius:8px;font-size:13px;margin-top:6px;">'+checklistAtual.obs+'</div></div>';}
  if(checklistAtual.assinatura_chefe_aprovado){
    html+='<div class="alert-s">✅ Este checklist já foi aprovado pelo chefe de equipe.</div>';
    document.getElementById('btn-assinar').style.display='none';
  }else{
    html+='<div style="margin-top:16px;"><div style="font-size:13px;font-weight:700;margin-bottom:8px;">Assinatura do Chefe</div><div class="sig-wrap"><canvas id="canvas-aprovacao"></canvas></div><button style="font-size:11px;color:var(--texto-sec);cursor:pointer;background:none;border:none;padding:4px 0;text-align:right;display:block;margin-top:4px;" onclick="clearSig()">Limpar assinatura</button></div>';
    document.getElementById('btn-assinar').style.display='block';
  }
  document.getElementById('modal-conteudo').innerHTML=html;
  document.getElementById('modal-checklist').classList.add('open');
  if(!checklistAtual.assinatura_chefe_aprovado)setTimeout(()=>initSig(),100);
}

function initSig(){
  const c=document.getElementById('canvas-aprovacao');
  if(!c)return;
  const ctx=c.getContext('2d');
  const dpr=window.devicePixelRatio||1;
  c.width=c.offsetWidth*dpr;c.height=120*dpr;
  ctx.scale(dpr,dpr);
  ctx.strokeStyle='#E8520A';ctx.lineWidth=2;ctx.lineCap='round';ctx.lineJoin='round';
  let drawing=false,lx,ly;
  function pos(e){const r=c.getBoundingClientRect(),s=e.touches?e.touches[0]:e;return[s.clientX-r.left,s.clientY-r.top];}
  c.onmousedown=e=>{drawing=true;[lx,ly]=pos(e);};
  c.ontouchstart=e=>{e.preventDefault();drawing=true;[lx,ly]=pos(e);};
  c.onmousemove=e=>{if(!drawing)return;const[x,y]=pos(e);ctx.beginPath();ctx.moveTo(lx,ly);ctx.lineTo(x,y);ctx.stroke();[lx,ly]=[x,y];};
  c.ontouchmove=e=>{e.preventDefault();if(!drawing)return;const[x,y]=pos(e);ctx.beginPath();ctx.moveTo(lx,ly);ctx.lineTo(x,y);ctx.stroke();[lx,ly]=[x,y];};
  ['mouseup','mouseleave','touchend'].forEach(ev=>c.addEventListener(ev,()=>drawing=false));
}

function clearSig(){const c=document.getElementById('canvas-aprovacao');if(c)c.getContext('2d').clearRect(0,0,c.width,c.height);}

async function assinarChecklist(){
  const btn=document.getElementById('btn-assinar');
  btn.disabled=true;btn.innerHTML='<span class="spinner"></span>Salvando...';
  const c=document.getElementById('canvas-aprovacao');
  const sig=c?c.toDataURL():'';
  try{
    const r=await fetch('/api/checklists/'+checklistAtual.id+'/aprovar',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({chefe:CHEFES[equipeLogada],equipe:equipeLogada,assinatura_chefe_aprovado:sig,aprovado_em:new Date().toISOString()})
    });
    if(!r.ok)throw new Error(await r.text());
    fecharModal();carregarChecklists();alert('✅ Checklist aprovado com sucesso!');
  }catch(e){alert('Erro: '+e.message);}
  finally{btn.disabled=false;btn.innerHTML='✍️ Assinar e Aprovar';}
}

function fecharModal(){document.getElementById('modal-checklist').classList.remove('open');checklistAtual=null;}
function sair(){equipeLogada=null;document.getElementById('tela-login').style.display='block';document.getElementById('tela-principal').style.display='none';document.getElementById('hdr-info').style.display='none';document.getElementById('login-equipe').value='';document.getElementById('login-senha').value='';}
document.getElementById('modal-checklist').addEventListener('click',e=>{if(e.target===e.currentTarget)fecharModal();});
document.getElementById('login-senha').addEventListener('keypress',e=>{if(e.key==='Enter')fazerLogin();});
</script>
